console.log("Global Namespace");

// Default root namespace in a browser is window
console.log(window);

// Each variable declared is attached in the window
var newVariable = "Hello World";
console.log(window.newVariable);

// The way to avoid collision is create variables in it's own scope (functions...)

function myScope() {
    var newName = "Lucas"
    return newName
}

console.log(myScope())
console.log(window.newName); //should be undefined

// Creating this variable should change that
var newName = "Lucas Fernandez";
console.log(window.newName); //should be Lucas Fernandez
